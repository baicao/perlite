// 测试7: 测试复杂数组操作（可能的崩溃原因）

DECLARE Matrix : ARRAY[1:3, 1:3] OF REAL
DECLARE TestArray : ARRAY[1:10] OF INTEGER
DECLARE row, col, i : INTEGER

// 二维数组操作
FOR row ← 1 TO 3
    FOR col ← 1 TO 3
        Matrix[row, col] ← row + col
    NEXT col
NEXT row

// 输出二维数组
OUTPUT "Matrix contents:"
FOR row ← 1 TO 3
    FOR col ← 1 TO 3
        OUTPUT Matrix[row, col], " "
    NEXT col
    OUTPUT ""
NEXT row

// 初始化一维数组
FOR i ← 1 TO 10
    TestArray[i] ← 11 - i
NEXT i

// 输出一维数组
OUTPUT "TestArray contents:"
FOR i ← 1 TO 10
    OUTPUT TestArray[i], " "
NEXT i

// 简单的加法函数
FUNCTION Add(a : INTEGER, b : INTEGER) RETURNS INTEGER
    RETURN a + b
ENDFUNCTION

// 计算平方的函数
FUNCTION Square(x : REAL) RETURNS REAL
    RETURN x * x
ENDFUNCTION

// 判断是否为偶数的函数
FUNCTION IsEven(num : INTEGER) RETURNS BOOLEAN
    IF num MOD 2 = 0 THEN
        RETURN TRUE
    ELSE
        RETURN FALSE
    ENDIF
ENDFUNCTION

// 基本 ARRAY 测试
// 测试 Cambridge 9618 Pseudocode ARRAY 语法

// 声明一维数组
DECLARE Numbers : ARRAY[1:5] OF INTEGER
DECLARE Names : ARRAY[0:2] OF STRING

// 数组赋值
Numbers[1] ← 10
Numbers[2] ← 20
Numbers[3] ← 30
Numbers[4] ← 40
Numbers[5] ← 50

Names[0] ← "Apple"
Names[1] ← "Banana"
Names[2] ← "Cherry"

// 使用循环操作数组
FOR i ← 1 TO 5
    OUTPUT "Numbers[", i, "] = ", Numbers[i]
NEXT i

FOR i ← 0 TO 2
    OUTPUT "Names[", i, "] = ", Names[i]
NEXT i

// 二维数组测试
DECLARE Matrix : ARRAY[1:2, 1:2] OF INTEGER

Matrix[1, 1] ← 1
Matrix[1, 2] ← 2
Matrix[2, 1] ← 3
Matrix[2, 2] ← 4

FOR row ← 1 TO 2
    FOR col ← 1 TO 2
        OUTPUT "Matrix[", row, ",", col, "] = ", Matrix[row, col]
    NEXT col
NEXT row
